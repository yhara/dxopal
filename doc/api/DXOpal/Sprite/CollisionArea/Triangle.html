def collides?(other)
  case other.type
  when :Point, :Circle, :Rect
    other.collides?(self)
  when :Triangle
    (ox1, oy1), (ox2, oy2), (ox3, oy3) = *self.absolute_poss
    (dx1, dy1), (dx2, dy2), (dx3, dy3) = *other.absolute_poss
    `Opal.DXOpal.CCk.check_triangle_triangle(ox1, oy1, ox2, oy2, ox3, oy3,
                                             dx1, dy1, dx2, dy2, dx3, dy3)`
  else raise
  end
end
